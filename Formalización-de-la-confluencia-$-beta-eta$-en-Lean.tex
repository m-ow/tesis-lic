\chapter{Relaciones}\label{relaciones}

\section{1.1. Definiciones básicas}\label{definiciones-buxe1sicas}

\begin{ndfn}{Definición 1.1.1}
Una relación binaria sobre un tipo \texttt{α} es una función
\texttt{α\ →\ α\ →\ Prop}, es decir, una proposición sobre pares de
elementos de \texttt{α}.

\end{ndfn}

\begin{Shaded}
\begin{Highlighting}[]
\NormalTok{variable \{α : }\KeywordTok{Type}\NormalTok{\} \{r : α → α → }\KeywordTok{Prop}\NormalTok{\}}
\end{Highlighting}
\end{Shaded}

\begin{ndfn}{Definición 1.1.2}

Sea \texttt{r} una relación en \texttt{α}. Entonces \texttt{r} induce
las siguientes relaciones binarias.

I. \texttt{ReflGen\ r} es la cerradura reflexiva de \texttt{r}, definida
inductivamente de la siguiente manera:

\begin{Shaded}
\begin{Highlighting}[]
\NormalTok{inductive ReflGen \{α : }\KeywordTok{Type}\NormalTok{\} (r : α → α → }\KeywordTok{Prop}\NormalTok{) : α → α → }\KeywordTok{Prop}
\NormalTok{  | refl \{a : α\}: ReflGen r a a}
\NormalTok{  | single \{a b : α\} : r a b → ReflGen r a b}
\end{Highlighting}
\end{Shaded}

II. \texttt{ReflTransGen\ r} es la cerradura reflexiva y transitiva de
\texttt{r}, definida inductivamente de la siguiente manera:

\begin{Shaded}
\begin{Highlighting}[]
\NormalTok{inductive ReflTransGen \{α : }\KeywordTok{Type}\NormalTok{\} (r : α → α → }\KeywordTok{Prop}\NormalTok{) : α → α → }\KeywordTok{Prop}
\NormalTok{  | refl \{a : α\} : ReflTransGen r a a}
\NormalTok{  | tail \{a b c : α\} : ReflTransGen r a b → r b c → ReflTransGen r a c}
\end{Highlighting}
\end{Shaded}

\end{ndfn}

\begin{nthm}{Teorema 1.1.1}

Si dos términos están relacionados por la cerradura reflexiva de
\texttt{r}, entonces también lo están por su cerradura reflexiva y
transitiva.

\begin{Shaded}
\begin{Highlighting}[]
\KeywordTok{theorem}\NormalTok{ ReflGen.to\_reflTransGen :}
\NormalTok{  ∀ \{a b : α\}, ReflGen r a b → ReflTransGen r a b}
\end{Highlighting}
\end{Shaded}

\end{nthm}

Las relaciones anteriores pueden describirse mediante caminos de
\texttt{a} a \texttt{b}:

\begin{itemize}
\tightlist
\item
  \texttt{r\ a\ b} indica la existencia de un camino de longitud 1.
\item
  \texttt{ReflGen\ r\ a\ b}, de un camino de longitud 0 o 1.
\item
  \texttt{ReflTransGen\ r\ a\ b}, de un camino de longitud \(n\), para
  algún \(n \in \mathbb{N}\).
\end{itemize}

Representamos estas relaciones mediante \texttt{a} \(\to\) \texttt{b},
\texttt{a} \(\to^{=}\) \texttt{b} y \texttt{a} \(\twoheadrightarrow\)
\texttt{b}, respectivamente.

\begin{ndfn}{Definición 1.1.3}

Sea \texttt{r} una relación en \texttt{α}.

Decimos que \texttt{r} tiene la \emph{propiedad del diamante} si, para
cualesquiera \texttt{a}, \texttt{b} y \texttt{c} tales que \texttt{a}
\(\to\) \texttt{b} y \texttt{a} \(\to\) \texttt{c}, existe un elemento
\texttt{d} tal que \texttt{b} \(\to\) \texttt{d} y \texttt{c} \(\to\)
\texttt{d}. Visualmente, esta propiedad afirma que el siguiente diagrama
siempre puede completarse:

\begin{center}
\begin{tikzpicture}
  \node (a) at (0, 2) {\lean{a}};
  \node (b) at (2, 2) {\lean{b}};
  \node (c) at (0, 0) {\lean{c}};
  \node (d) at (2, 0) {\lean{d}};
  \draw[->]           (a) -- node[above] {\lean{r}} (b);
  \draw[->]           (a) -- node[left]  {\lean{r}} (c);
  \draw[->, dashed] (b) -- node[right] {\lean{r}} (d);
  \draw[->, dashed] (c) -- node[below] {\lean{r}} (d);
\end{tikzpicture}
\end{center}

\begin{Shaded}
\begin{Highlighting}[]
\KeywordTok{def}\NormalTok{ Diamond (r : α → α → }\KeywordTok{Prop}\NormalTok{) :=}
\NormalTok{  ∀ \{a b c : α\}, r a b → r a c → ∃ d, r b d ∧ r c d}
\end{Highlighting}
\end{Shaded}

\end{ndfn}

\subsection{Confluencia y
Conmutación}\label{confluencia-y-conmutaciuxf3n}

\begin{ndfn}{Definición 1.1.4}

Sea \texttt{r} una relación en \texttt{α}.

I. \texttt{r} es \textbf{confluente} si su cerradura reflexiva y
transitiva tiene la propiedad del diamante.

\begin{center}
\begin{tikzpicture}
  \node (a) at (0, 2) {\lean{a}};
  \node (b) at (2, 2) {\lean{b}};
  \node (c) at (0, 0) {\lean{c}};
  \node (d) at (2, 0) {\lean{d}};
  \draw[->>]           (a) -- node[above] {\lean{r}} (b);
  \draw[->>]           (a) -- node[left]  {\lean{r}} (c);
  \draw[->>, dashed] (b) -- node[right] {\lean{r}} (d);
  \draw[->>, dashed] (c) -- node[below] {\lean{r}} (d);
\end{tikzpicture}
\end{center}

\begin{Shaded}
\begin{Highlighting}[]
\KeywordTok{def}\NormalTok{ Confluent (r : α → α → }\KeywordTok{Prop}\NormalTok{) := Diamond (ReflTransGen r)}
\end{Highlighting}
\end{Shaded}

II. \texttt{r} es \textbf{fuertemente confluente} si:

\begin{center}
\begin{tikzpicture}
  \node (a) at (0, 2) {\lean{x}};
  \node (b) at (2, 2) {\lean{y₁}};
  \node (c) at (0, 0) {\lean{y₂}};
  \node (d) at (2, 0) {\lean{z}};
  \draw[->]         (a) -- node[above] {\lean{r}} (b);
  \draw[->]         (a) -- node[left]  {\lean{r}} (c);
  \draw[->, dashed] (b) -- node[right] {\lean{r}} node[left] {$=$} (d);
  \draw[->>, dashed] (c) -- node[below] {\lean{r}} (d);
\end{tikzpicture}
\end{center}

\begin{Shaded}
\begin{Highlighting}[]
\KeywordTok{def}\NormalTok{ StronglyConfluent (r : α → α → }\KeywordTok{Prop}\NormalTok{) :=}
\NormalTok{  ∀ \{x y₁ y₂\}, r x y₁ → r x y₂ → ∃ z, ReflGen r y₁ z ∧ ReflTransGen r y₂ z}
\end{Highlighting}
\end{Shaded}

\end{ndfn}

\begin{nthm}{Teorema 1.1.2}

Si una relación es fuertemente confluente, entonces también es
confluente.

\begin{Shaded}
\begin{Highlighting}[]
\KeywordTok{theorem}\NormalTok{ StronglyConfluent.toConfluent (h : StronglyConfluent r) : Confluent r }
\end{Highlighting}
\end{Shaded}

\end{nthm}

\begin{ndfn}{Definición 1.1.5}

Sean \texttt{r₁} y \texttt{r₂} dos relaciones en \texttt{α}.

I. \texttt{r₁} y \texttt{r₂} \textbf{conmutan} si:

\begin{center}
\begin{tikzpicture}
  \node (A) at (0, 2) {\lean{x}};
  \node (B) at (2, 2) {\lean{y₁}};
  \node (C) at (0, 0) {\lean{y₂}};
  \node (D) at (2, 0) {\lean{z}};
  \draw[->>]         (A) -- node[above] {\lean{r₁}} (B);
  \draw[->>]         (A) -- node[left]  {\lean{r₂}} (C);
  \draw[->>, dashed] (B) -- node[right] {\lean{r₂}} (D);
  \draw[->>, dashed] (C) -- node[below] {\lean{r₁}} (D);
\end{tikzpicture}
\end{center}

\begin{Shaded}
\begin{Highlighting}[]
\KeywordTok{def}\NormalTok{ Commute (r₁ r₂ : α → α → }\KeywordTok{Prop}\NormalTok{) :=}
\NormalTok{  ∀ \{x y₁ y₂\}, ReflTransGen r₁ x y₁ → ReflTransGen r₂ x y₂ → }
\NormalTok{    ∃ z, ReflTransGen r₂ y₁ z ∧ ReflTransGen r₁ y₂ z}
\end{Highlighting}
\end{Shaded}

II. \texttt{r₁} y \texttt{r₂} \textbf{conmutan fuertemente} si:

\begin{center}
\begin{tikzpicture}
  \node (A) at (0, 2) {\lean{x}};
  \node (B) at (2, 2) {\lean{y₁}};
  \node (C) at (0, 0) {\lean{y₂}};
  \node (D) at (2, 0) {\lean{z}};
  \draw[->] (A) -- node[above] {\lean{r₁}}  (B);
  \draw[->] (A) -- node[left] {\lean{r₂}} (C);
  \draw[->, dashed] (B)-- node[right] {\lean{r₂}} node[left] {$=$} (D);
  \draw[->>, dashed] (C)-- node[below]  {\lean{r₁}}  (D);
\end{tikzpicture}
\end{center}

\begin{Shaded}
\begin{Highlighting}[]
\KeywordTok{def}\NormalTok{ StronglyCommute (r₁ r₂ : α → α → }\KeywordTok{Prop}\NormalTok{) :=}
\NormalTok{  ∀ \{x y₁ y₂\}, r₁ x y₁ → r₂ x y₂ → ∃ z, ReflGen r₂ y₁ z ∧ ReflTransGen r₁ y₂ z}
\end{Highlighting}
\end{Shaded}

\end{ndfn}

\begin{nthm}{Teorema 1.1.3}

Si dos relaciones conmutan fuertemente, entonces también conmutan.

\begin{Shaded}
\begin{Highlighting}[]
\KeywordTok{theorem}\NormalTok{ StronglyCommute.toCommute \{r₁ r₂ : α → α → }\KeywordTok{Prop}\NormalTok{\} }
\NormalTok{  (h : StronglyCommute r₁ r₂) : Commute r₁ r₂ }
\end{Highlighting}
\end{Shaded}

\end{nthm}

\begin{nthm}{Teorema 1.1.4}

La conmutación entre relaciones es simétrica, es decir, si \texttt{r₁} y
\texttt{r₂} conmutan, entonces \texttt{r₂} y \texttt{r₁} también.

\begin{Shaded}
\begin{Highlighting}[]
\KeywordTok{theorem}\NormalTok{ Commute.symmetric : Symmetric (@Commute α)}
\end{Highlighting}
\end{Shaded}

\end{nthm}

\begin{nthm}{Teorema 1.1.5 (Hindley-Rosen)}

Sean \texttt{r₁} y \texttt{r₂} dos relaciones sobre un mismo tipo. Si
ambas son confluentes y conmutan entre sí, entonces su unión
\texttt{r₁\ ⊔\ r₂} es confluente.

\NormalTok{  (c₁ : Confluent r₁) (c₂ : Confluent r₂) (comm : Commute r₁ r₂) : }
\NormalTok{    Confluent (r₁ ⊔ r₂)}
\end{Highlighting}
\end{Shaded}

\end{nthm}

\section{\texorpdfstring{2.1. La representación \emph{locally
nameless}}{2.1. La representación locally nameless}}\label{la-representaciuxf3n-locally-nameless}

\begin{ndfn}{Definición 2.1.1}
Utilizamos una representación \emph{locally nameless} para el cálculo
\(\lambda\), donde las variables ligadas se representan como números
naturales (índices de De Bruijn) y las variables libres como términos
del tipo \texttt{Var}. Sobre el tipo \texttt{Var} la igualdad es
decidible (\texttt{DecidableEq}) y es posible generar una variable fuera
de cualquier conjunto finito dado (\texttt{HasFresh}).

\begin{Shaded}
\begin{Highlighting}[]
\NormalTok{universe u}
\NormalTok{variable \{Var : }\KeywordTok{Type}\NormalTok{ u\} [HasFresh Var] [DecidableEq Var]}

\NormalTok{inductive Term (Var : }\KeywordTok{Type}\NormalTok{ u)}
\NormalTok{  | bvar : ℕ → Term Var}
\NormalTok{  | fvar : Var → Term Var}
\NormalTok{  | abs  : Term Var → Term Var}
\NormalTok{  | app  : Term Var → Term Var → Term Var}
\end{Highlighting}
\end{Shaded}

Por ejemplo, podemos representar la expresión
``\(\lambda x. \, x \, y\)'' de la siguiente manera.

\begin{Shaded}
\begin{Highlighting}[]
\NormalTok{variable \{y : Var\}}
\KeywordTok{def}\NormalTok{ ejemplo\_rep := abs (app (bvar 0) (fvar y))}
\end{Highlighting}
\end{Shaded}

Lo cual también se puede escribir como ``\(\lambda . \, 0 \, y\)''.

\end{ndfn}

\begin{ndfn}{Definición 2.1.2}

La sustitución consiste en remplazar una variable libre por un término.

\begin{Shaded}
\begin{Highlighting}[]
\KeywordTok{def}\NormalTok{ subst (m : Term Var) (x : Var) (sub : Term Var) : Term Var :=}
  \KeywordTok{match}\NormalTok{ m }\KeywordTok{with}
\NormalTok{  | bvar i  =\textgreater{} bvar i}
\NormalTok{  | fvar x\textquotesingle{} =\textgreater{} if x = x\textquotesingle{} then sub else fvar x\textquotesingle{}}
\NormalTok{  | app l r =\textgreater{} app (subst l x sub) (subst r x sub)}
\NormalTok{  | abs M   =\textgreater{} abs (subst M x sub)}
\end{Highlighting}
\end{Shaded}

Definimos una notación para la sustitución de variables libres que imita
la notación matemática estándar.

\begin{Shaded}
\begin{Highlighting}[]
\NormalTok{scoped }\KeywordTok{notation}\NormalTok{ t:max }\StringTok{"["}\NormalTok{ x }\StringTok{":="}\NormalTok{ t\textquotesingle{} }\StringTok{"]"}\NormalTok{ =\textgreater{} subst t x t\textquotesingle{}}
\end{Highlighting}
\end{Shaded}

\end{ndfn}

\begin{ndfn}{Definición 2.1.3}
La función \texttt{fv}, definida a continuación, calcula el conjunto de
variables libres de una expresión. Dado que utilizamos una
representación \emph{locally nameless}, donde las variables ligadas se
representan como índices, cualquier nombre que veamos es una variable
libre de un término.

\begin{Shaded}
\begin{Highlighting}[]
\KeywordTok{def}\NormalTok{ fv : Term Var → Finset Var}
\NormalTok{  | bvar \_ =\textgreater{} \{\}}
\NormalTok{  | fvar x =\textgreater{} \{x\}}
\NormalTok{  | abs e1 =\textgreater{} fv e1}
\NormalTok{  | app l r =\textgreater{} fv l ∪ fv r}
\end{Highlighting}
\end{Shaded}

El tipo \texttt{Finset\ Var} representa un conjunto finito de elementos
del tipo \texttt{Var}.

\end{ndfn}

\begin{ndfn}{Definición 2.1.4}
La operación de apertura sustituye un índice por un término.

\begin{Shaded}
\begin{Highlighting}[]
\KeywordTok{def}\NormalTok{ openRec (i : ℕ) (sub : Term Var) : Term Var → Term Var}
\NormalTok{  | bvar i\textquotesingle{} =\textgreater{} if i = i\textquotesingle{} then sub else bvar i\textquotesingle{}}
\NormalTok{  | fvar x  =\textgreater{} fvar x}
\NormalTok{  | app l r =\textgreater{} app (openRec i sub l) (openRec i sub r)}
\NormalTok{  | abs M   =\textgreater{} abs (openRec (i+1) sub M)}
\end{Highlighting}
\end{Shaded}

Su notación es la siguiente.

\begin{Shaded}
\begin{Highlighting}[]
\NormalTok{scoped }\KeywordTok{notation}\NormalTok{:68 e }\StringTok{"⟦"}\NormalTok{ i }\StringTok{" ↝ "}\NormalTok{ sub }\StringTok{"⟧"}\NormalTok{ =\textgreater{} openRec i sub e}
\end{Highlighting}
\end{Shaded}

Las aplicaciones habituales de esta función sustituyen el índice cero
por una variable o un término \(\lambda\). La siguiente definición es
una forma práctica de indicar estos usos.

\begin{Shaded}
\begin{Highlighting}[]
\KeywordTok{def}\NormalTok{ open\textquotesingle{} \{X\} (e u) := @openRec X 0 u e  }
 
\NormalTok{scoped infixr:80 }\StringTok{" \^{} "}\NormalTok{ =\textgreater{} open\textquotesingle{}}
\end{Highlighting}
\end{Shaded}

Así, \texttt{(e\ \^{}\ x)} puede leerse como ``reemplaza el índice \(0\)
por la variable \texttt{x} en \texttt{e}'' o ``abre \texttt{e} con la
variable \texttt{x}''.

\end{ndfn}

\begin{ndfn}{Definición 2.1.5}

La operación de cierre reemplaza la variable libre \texttt{x} por el
índice \texttt{k}. Es la operación inversa de la apertura.

\begin{Shaded}
\begin{Highlighting}[]
\KeywordTok{def}\NormalTok{ closeRec (k : ℕ) (x : Var) : Term Var → Term Var}
\NormalTok{  | fvar x\textquotesingle{} =\textgreater{} if x = x\textquotesingle{} then bvar k else fvar x\textquotesingle{}}
\NormalTok{  | bvar i  =\textgreater{} bvar i}
\NormalTok{  | app l r =\textgreater{} app (closeRec k x l) (closeRec k x r)}
\NormalTok{  | abs t   =\textgreater{} abs (closeRec (k+1) x t)}
\end{Highlighting}
\end{Shaded}

Cuya notación es la siguiente.

\begin{Shaded}
\begin{Highlighting}[]
\NormalTok{scoped }\KeywordTok{notation}\NormalTok{:68 e }\StringTok{"⟦"}\NormalTok{ k }\StringTok{" ↜ "}\NormalTok{ x }\StringTok{"⟧"}\NormalTok{=\textgreater{} closeRec k x e}
\end{Highlighting}
\end{Shaded}

Al igual que con la apertura, el uso habitual es sobre el índice cero.
La siguiente definición trata ese caso.

\begin{Shaded}
\begin{Highlighting}[]
\KeywordTok{def}\NormalTok{ close (e u) := @closeRec Var \_ 0 u e}

\NormalTok{scoped infixr:80 }\StringTok{" \^{}* "}\NormalTok{ =\textgreater{} close}
\end{Highlighting}
\end{Shaded}

\end{ndfn}

\begin{nthm}{Teorema 2.1.1}

Abrir un término con una variable que no aparece libre en él y después
cerrar sobre esa misma variable devuelve el término original.

\begin{Shaded}
\begin{Highlighting}[]
\KeywordTok{theorem}\NormalTok{ open\_close\_var (x : Var) (M : Term Var) (h : x ∉ fv M) : }
\NormalTok{  M = (M \^{} fvar x) \^{}* x}
\end{Highlighting}
\end{Shaded}

\end{nthm}

\begin{ndfn}{Definición 2.1.6 (Local closure)}

Un término se denomina ``localmente cerrado'' cuando no contiene índices
no ligados. La proposición \texttt{LC\ e} se cumple cuando una expresión
\texttt{e} es localmente cerrada.

\begin{Shaded}
\begin{Highlighting}[]
\NormalTok{inductive LC : Term Var → }\KeywordTok{Prop}
\NormalTok{  | fvar (x : Var)  : LC (fvar x)}
\NormalTok{  | abs (L : Finset Var) (e : Term Var) : (∀ x ∉ L, LC (e \^{} fvar x)) → LC (abs e)}
\NormalTok{  | app \{l r\} : l.LC → r.LC → LC (app l r)}
\end{Highlighting}
\end{Shaded}

\end{ndfn}

\begin{ndfn}{Definición 2.1.7}

Dada una relación \texttt{r} en el tipo \texttt{Term\ Var}, se llama
cerradura compatible de \texttt{r} a la relación \texttt{CompatGen\ r},
definida de la siguiente manera.

\begin{Shaded}
\begin{Highlighting}[]
\NormalTok{inductive CompatGen (r : Term Var → Term Var → }\KeywordTok{Prop}\NormalTok{) : Term Var → Term Var → }\KeywordTok{Prop}
\NormalTok{  | base : r M N → CompatGen r M N}
\NormalTok{  | appL : LC Z → CompatGen r M N → CompatGen r (app M Z) (app N Z)}
\NormalTok{  | appR: LC Z → CompatGen r M N → CompatGen r (app Z M) (app Z N)}
\NormalTok{  | xi (L : Finset Var) : (∀ x ∉ L, CompatGen r (M \^{} fvar x) (N \^{} fvar x)) →}
\NormalTok{    CompatGen r (abs M) (abs N)}
\end{Highlighting}
\end{Shaded}

\end{ndfn}

Sea \texttt{Beta} una relación en \texttt{Term\ Var}, definida de la

\begin{Shaded}
\begin{Highlighting}[]
\NormalTok{inductive Beta : Term Var → Term Var → }\KeywordTok{Prop}

\end{Highlighting}
\end{Shaded}

\end{Shaded}

\end{ndfn}

\begin{ndfn}{Definición 2.1.9}
Sea \texttt{Eta} una relación en \texttt{Term\ Var}, definida de la

\begin{Shaded}

\subsection{Elección de variables
nuevas}\label{elecciuxf3n-de-variables-nuevas}

Llamamos \emph{nueva} a una variable que todavía no se usa en la
demostración. Al pedir en la definición 2.1.1 que pueda generarse una
variable fuera de cualquier conjunto finito, nos aseguramos de que
siempre haya una variable nueva disponible cuando se necesite. El
teorema \texttt{fresh\_exists} enuncia esta propiedad:

\begin{Shaded}
\begin{Highlighting}[]
\KeywordTok{theorem}\NormalTok{ fresh\_exists (L : Finset Var) : ∃ x, x ∉ L}
\end{Highlighting}
\end{Shaded}

El conjunto al que lo aplicamos lo construye
\texttt{free\_union\ {[}fv{]}\ Var}: la unión de todas las variables,
todos los conjuntos finitos y las variables libres de todos los términos
presentes en la demostración.

Por ejemplo, supongamos que en una demostración figuran una variable
\texttt{x}, un conjunto finito \texttt{L} y un término \texttt{M}. La
siguiente línea aplica el teorema anterior:

\begin{Shaded}
\begin{Highlighting}[]
\KeywordTok{have}\NormalTok{ ⟨w, hw⟩ := fresh\_exists (free\_union [fv] Var)}
\end{Highlighting}
\end{Shaded}

Esto produce una variable \texttt{w} junto con la hipótesis:

\begin{Shaded}
\begin{Highlighting}[]
\InformationTok{hw }\NormalTok{: w ∉ ∅ ∪ \{x\} ∪ L ∪ fv M}
\end{Highlighting}
\end{Shaded}

En adelante, cuando escribamos ``sea \texttt{w} una variable nueva'',
nos referimos a esta construcción. Las condiciones particulares que se
necesiten después (\texttt{w\ ∉\ L}, \texttt{w\ ∉\ fv\ M}, etcétera) son
consecuencias inmediatas de \texttt{hw}, y las resuelve la táctica
\texttt{aesop} (\emph{Automated Extensible Search for Obvious Proofs}),
un procedimiento que busca demostraciones de manera automática.

\section{\texorpdfstring{3.1. Confluencia fuerte de

\begin{nthm}{Teorema 3.1.1}
La reducción \(\eta\) de un solo paso es fuertemente confluente.

\end{nthm}

\begin{proof}
Supongamos que \texttt{M}, \texttt{M₁} y \texttt{M₂} son términos lambda
tales que \texttt{M\ ⭢ηᶠ\ M₁} y \texttt{M\ ⭢ηᶠ\ M₂}. Entonces existe un
término lambda \texttt{z} tal que \texttt{ReflGen\ FullEta\ M₁\ z} y
\texttt{M₂\ ↠ηᶠ\ z}.

\begin{inpt}

\begin{Shaded}
\begin{Highlighting}[]
\KeywordTok{theorem}\NormalTok{ strong\_confluence\_eta :}
\NormalTok{    StronglyConfluent (@FullEta Var) := }\KeywordTok{by}
\NormalTok{  intro M M₁ M₂ h₁ h₂}
\end{Highlighting}
\end{Shaded}

\end{inpt}

\begin{outpt}

\begin{Shaded}
\begin{Highlighting}[]
\InformationTok{M M₁ M₂ }\NormalTok{: Term Var}
\InformationTok{h₁ }\NormalTok{: M ⭢ηᶠ M₁}
\InformationTok{h₂ }\NormalTok{: M ⭢ηᶠ M₂}
\NormalTok{⊢ ∃ z, ReflGen FullEta M₁ z }
\SpecialCharTok{\textgreater{}\textgreater{}}\NormalTok{    ∧ M₂ ↠ηᶠ z}
\end{Highlighting}
\end{Shaded}

\end{outpt}

Por el teorema 1.1.1 (\texttt{ReflGen.to\_reflTransGen}), basta
encontrar un término \texttt{z} tal que \texttt{ReflGen\ FullEta\ M₁\ z}
y \texttt{ReflGen\ FullEta\ M₂\ z}.

\begin{inpt}

\begin{Shaded}
\begin{Highlighting}[]
\KeywordTok{theorem}\NormalTok{ strong\_confluence\_eta :}
\NormalTok{    StronglyConfluent (@FullEta Var) := }\KeywordTok{by}
\NormalTok{  intro M M₁ M₂ h₁ h₂}
\NormalTok{  suffices h : ∃ z,}
\NormalTok{      ReflGen FullEta M₁ z ∧ ReflGen FullEta M₂ z }\KeywordTok{by}
    \KeywordTok{have}\NormalTok{ ⟨n, hn₁, hn₂⟩ := h}
\NormalTok{    use n }\CommentTok{{-}{-} ⊢}
\NormalTok{    constructor}
\NormalTok{    · }\KeywordTok{exact}\NormalTok{ hn₁}
\NormalTok{    · }\KeywordTok{apply}\NormalTok{ ReflGen.to\_reflTransGen}
      \KeywordTok{exact}\NormalTok{ hn₂}
\NormalTok{  induction h₁ generalizing M₂}
\end{Highlighting}
\end{Shaded}

\end{inpt}

\begin{outpt}

\begin{Shaded}
\begin{Highlighting}[]
\InformationTok{M M₁ M₂ n }\NormalTok{: Term Var}
\InformationTok{h₁ }\NormalTok{: M ⭢ηᶠ M₁}
\InformationTok{h₂ }\NormalTok{: M ⭢ηᶠ M₂}
\InformationTok{h }\NormalTok{: ∃ z, }
\SpecialCharTok{\textgreater{}\textgreater{}}\NormalTok{    ReflGen FullEta M₁ z }
\SpecialCharTok{\textgreater{}\textgreater{}}\NormalTok{    ∧ ReflGen FullEta M₂ z}
\InformationTok{hn₁ }\NormalTok{: ReflGen FullEta M₁ n}
\InformationTok{hn₂ }\NormalTok{: ReflGen FullEta M₂ n}
\NormalTok{⊢ ReflGen FullEta M₁ n }
\SpecialCharTok{\textgreater{}\textgreater{}}\NormalTok{    ∧ M₂ ↠ηᶠ n}
\end{Highlighting}
\end{Shaded}

\end{outpt}

Encontraremos dicho término por inducción sobre la derivación de
\texttt{h₁\ :\ M\ ⭢ηᶠ\ M₁}.

Demostraremos que, para todo término \texttt{M₂}, si
\texttt{M\ ⭢ηᶠ\ M₂}, entonces existe un término \texttt{z} tal que
\texttt{ReflGen\ FullEta\ M₁\ z} y \texttt{ReflGen\ FullEta\ M₂\ z}.
Procedemos por casos según la última regla empleada en la derivación de
\texttt{h₁}.

\textbf{I}. Si la última regla fue \texttt{base}, entonces \texttt{M}
\(=\) \texttt{abs\ (app\ P\ (bvar\ 0))} y \texttt{M₁} \(=\) \texttt{P},
para alguna \texttt{P} tal que \texttt{LC\ P}.

\begin{Shaded}
\begin{Highlighting}[]
\KeywordTok{theorem}\NormalTok{ strong\_confluence\_eta :}
\NormalTok{    StronglyConfluent (@FullEta Var) := }\KeywordTok{by}
\NormalTok{  intro M M₁ M₂ h₁ h₂}
\NormalTok{  ⋮}
\NormalTok{  case base \_ \_ P\textquotesingle{} P e\_P =\textgreater{}}
    \KeywordTok{have}\NormalTok{ ⟨lc\_P⟩ := e\_P}
    \KeywordTok{have}\NormalTok{ h := invert\_step\_eta\_redex lc\_P h₂}
\end{Highlighting}
\end{Shaded}

\begin{Shaded}
\begin{Highlighting}[]
\InformationTok{M M₁ M₂ P\textquotesingle{} P }\NormalTok{: Term Var}
\InformationTok{lc\_P }\NormalTok{: LC P}
\InformationTok{e\_P }\NormalTok{: Eta P\textquotesingle{} P}
\InformationTok{h₂ }\NormalTok{: abs (app P (bvar 0)) ⭢ηᶠ M₂}
\InformationTok{h }\NormalTok{: M₂ = P ∨ ∃ N\textquotesingle{}, P ⭢ηᶠ N\textquotesingle{} ∧ M₂ = abs (app N\textquotesingle{} (bvar 0))}
\NormalTok{⊢ ∃ z, ReflGen FullEta P z ∧ ReflGen FullEta M₂ z}
\end{Highlighting}
\end{Shaded}

Aplicando el teorema 2.1.3 (\texttt{invert\_step\_eta\_redex}) a
\texttt{h₂} obtenemos dos posibilidades: \texttt{M₂} \(=\) \texttt{P}, o
bien \texttt{M₂} \(=\) \texttt{abs\ (app\ P₁\ (bvar\ 0))} para alguna
\texttt{P₁} con \texttt{P\ ⭢ηᶠ\ P₁}. Analizamos cada una.

\textbf{(a)}~Si \texttt{M₂} \(=\) P, ambas derivaciones aplicaron la
regla \(\eta\) a \texttt{abs\ (app\ P\ (bvar\ 0))} y llegaron al mismo
término. Basta tomar \texttt{z} \(=\) \texttt{P}.

\begin{inpt}

\begin{Shaded}
\begin{Highlighting}[]
\NormalTok{  case base \_ \_ P\textquotesingle{} P e\_P =\textgreater{}}
    \KeywordTok{have}\NormalTok{ ⟨lc\_P⟩ := e\_P}
    \KeywordTok{have}\NormalTok{ h := invert\_step\_eta\_redex lc\_P h₂}
\NormalTok{    cases h }\KeywordTok{with}
\NormalTok{    | inl h\_eq =\textgreater{}}
        \KeywordTok{rw}\NormalTok{ [h\_eq] }\CommentTok{{-}{-} ⊢}
\NormalTok{        use P}
\end{Highlighting}
\end{Shaded}

\end{inpt}

\begin{outpt}

\begin{Shaded}
\begin{Highlighting}[]
\InformationTok{M M₁ M₂ P\textquotesingle{} P }\NormalTok{: Term Var}

Para probar el primer componente de la conjunción procedemos por casos
sobre \texttt{ReflGen\ FullEta\ (Q\ \^{}\ fvar\ w)\ P₂}:

\begin{itemize}
\item
  Si la reducción es reflexiva, entonces \texttt{P₂} \(=\)
  \texttt{Q\ \^{}\ fvar\ w}, de modo que \texttt{z} \(=\)
  \texttt{abs\ ((Q\ \^{}\ fvar\ w)\ \^{}*\ w)}. Por el teorema 2.1.1
  (\texttt{open\_close\_var}) este término es \texttt{abs\ Q} y ambos
  lados coinciden.
\item
  Si la reducción consta de un paso
  \texttt{st\ :\ (Q\ \^{}\ fvar\ w)\ ⭢ηᶠ\ P₂}, reescribimos
  \texttt{abs\ Q} como \texttt{abs\ ((Q\ \^{}\ fvar\ w)\ \^{}*\ w)}
  mediante el teorema 2.1.1 (\texttt{open\_close\_var}), así, basta
  construir la reducción
  \texttt{abs\ ((Q\ \^{}\ fvar\ w)\ \^{}*\ w)\ ⭢ηᶠ\ abs\ (P₂\ \^{}*\ w)}.

  De \texttt{h\_xi} aplicada a \texttt{w} obtenemos
  \texttt{(P\ \^{}\ fvar\ w)\ ⭢ηᶠ\ (Q\ \^{}\ fvar\ w)}, que garantiza
  \texttt{LC\ (Q\ \^{}\ fvar\ w)}, con esta premisa, el teorema 2.1.4
  (\texttt{step\_abs\_close}) aplicado a \texttt{st} produce la
  reducción buscada.
\end{itemize}
